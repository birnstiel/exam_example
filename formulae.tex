
% Pythagorean theorem
\begin{equation}
 a^2 + b^2 = c^2
\end{equation}

% Quadratic formula
\begin{equation}
 x = \frac{-b \pm \sqrt{b^2 - 4ac}}{2a}
\end{equation}

% hypergeometric series
\begin{equation}
  \sum_{n=0}^{\infty} \frac{1}{n!} = e
\end{equation}

% Euler's formula
\begin{equation}
  e^{i\theta} = \cos(\theta) + i\sin(\theta)
\end{equation}

% Euler's identity
\begin{equation}
  e^{i\pi} + 1 = 0
\end{equation}

% Fundamental theorem of calculus
\begin{equation}
  \int_a^b f'(x) \, dx = f(b) - f(a)
\end{equation}
